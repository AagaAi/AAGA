async function loadStrategyCompare() { const container = document.getElementById('strategyCompare'); container.innerHTML = 'Loading...'; try { const res = await fetch('/compare-strategies'); const data = await res.json(); if (data.error) { container.innerHTML = `${data.error}`; return; } let html = ''; data.results.forEach(r => { html += `
${r.strategy}: Trades ${r.trades}, Win Rate ${r.win_rate}%, ROI ${r.roi}%, Balance $${r.final_balance}
`; }); if (data.best_strategy) { html += `

🏆 Best: ${data.best_strategy}

`; } container.innerHTML = html || 'No data'; } catch(e) { container.innerHTML = 'Failed to load comparison'; console.error(e); } }
